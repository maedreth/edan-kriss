var Edan = require('./edan');
const Learn = require('./learn');
Edan.job = "Not what he loves";

while (Edan.job != "Developer") {
    solidifySkills(Edan.courses, Edan.projects);
};

function solidifySkills (courses, projects) {
    if (Edan.newIdea) {
        Edan.projects.push(Edan.newIdea);
        delete Edan.newIdea;
    };   
    Learn.onlineCourses(courses);
    Learn.developProjects(projects);
};
